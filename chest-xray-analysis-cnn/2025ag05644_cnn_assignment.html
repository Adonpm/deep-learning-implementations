class TransformDataset(Dataset):
    def __init__(self, dataset, transform):
        self.dataset   = dataset
        self.transform = transform

    def __len__(self):
        return len(self.dataset)

    def __getitem__(self, idx):
        image, label = self.dataset[idx]
        if self.transform:
            image = self.transform(image)
        return image, label

def imshow_grid(loader, class_names, n=8, title='Sample Images'):
    images, labels = next(iter(loader))
    images = images[:n]
    labels = labels[:n]

    # Denormalise for display
    mean = torch.tensor([0.485, 0.456, 0.406]).view(3,1,1)
    std  = torch.tensor([0.229, 0.224, 0.225]).view(3,1,1)
    images = images * std + mean
    images = images.clamp(0, 1)

    fig, axes = plt.subplots(1, n, figsize=(2*n, 2.5))
    for i, ax in enumerate(axes):
        img = images[i].permute(1,2,0).numpy()
        ax.imshow(img, cmap='gray')
        ax.set_title(class_names[labels[i].item()], fontsize=8)
        ax.axis('off')
    fig.suptitle(title, fontsize=12, fontweight='bold')
    plt.tight_layout()
    plt.show()

Part 2: Custom CNN Implementation¶

Architecture: 3 × Conv-BN-ReLU-MaxPool blocks → Global Average Pooling → FC output
GAP replaces Flatten+Dense to reduce parameters and overfitting (mandatory requirement).

class CustomCNN(nn.Module):
    """
    Custom CNN with Global Average Pooling.
    Architecture:
      Conv(3->32) -> BN -> ReLU -> MaxPool
      Conv(32->64) -> BN -> ReLU -> MaxPool
      Conv(64->128) -> BN -> ReLU -> MaxPool
      GlobalAveragePooling  (AdaptiveAvgPool2d(1))
      FC(128 -> n_classes)
    """
    def __init__(self, n_classes=2):
        super(CustomCNN, self).__init__()

        # Block 1
        self.conv1 = nn.Sequential(
            nn.Conv2d(3, 32, kernel_size=3, padding=1),
            nn.BatchNorm2d(32),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2)          # 224 -> 112
        )
        # Block 2
        self.conv2 = nn.Sequential(
            nn.Conv2d(32, 64, kernel_size=3, padding=1),
            nn.BatchNorm2d(64),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2)          # 112 -> 56
        )
        # Block 3
        self.conv3 = nn.Sequential(
            nn.Conv2d(64, 128, kernel_size=3, padding=1),
            nn.BatchNorm2d(128),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2)          # 56 -> 28
        )

        # MANDATORY: Global Average Pooling (NOT Flatten+Dense)
        self.gap = nn.AdaptiveAvgPool2d(1)   # (B, 128, 28, 28) -> (B, 128, 1, 1)

        # Classifier head
        self.classifier = nn.Linear(128, n_classes)

    def forward(self, x):
        x = self.conv1(x)
        x = self.conv2(x)
        x = self.conv3(x)
        x = self.gap(x)           # Global Average Pooling
        x = x.view(x.size(0), -1) # (B, 128)
        x = self.classifier(x)
        return x

# ── Training utilities ────────────────────────────────────────────────────────
def train_one_epoch(model, loader, criterion, optimizer, device):
    model.train()
    running_loss = 0.0
    for images, labels in loader:
        images, labels = images.to(device), labels.to(device)
        optimizer.zero_grad()
        outputs = model(images)
        loss    = criterion(outputs, labels)
        loss.backward()
        optimizer.step()
        running_loss += loss.item() * images.size(0)
    return running_loss / len(loader.dataset)

@torch.no_grad()
def evaluate(model, loader, criterion, device):
    model.eval()
    running_loss, all_preds, all_labels = 0.0, [], []
    for images, labels in loader:
        images, labels = images.to(device), labels.to(device)
        outputs = model(images)
        loss    = criterion(outputs, labels)
        running_loss += loss.item() * images.size(0)
        preds = outputs.argmax(dim=1)
        all_preds.extend(preds.cpu().numpy())
        all_labels.extend(labels.cpu().numpy())
    return running_loss / len(loader.dataset), np.array(all_preds), np.array(all_labels)

def compute_metrics(y_true, y_pred, average='macro'):
    acc  = accuracy_score(y_true, y_pred)
    prec = precision_score(y_true, y_pred, average=average, zero_division=0)
    rec  = recall_score(y_true, y_pred, average=average, zero_division=0)
    f1   = f1_score(y_true, y_pred, average=average, zero_division=0)
    return acc, prec, rec, f1

# ── Build Transfer Learning Model ─────────────────────────────────────────────
# ResNet18 already ends with AdaptiveAvgPool2d + FC.
# We replace the FC head and keep the built-in GAP.

def build_transfer_model(n_classes, freeze_base=True):
    # Load pretrained ResNet18
    base = models.resnet18(weights=models.ResNet18_Weights.IMAGENET1K_V1)

    # Freeze all base layers
    if freeze_base:
        for param in base.parameters():
            param.requires_grad = False

    # ResNet18 already has: ...layer4 -> AdaptiveAvgPool2d(1) -> Flatten -> fc
    # We only replace the final FC layer
    # The AdaptiveAvgPool2d(1) is the Global Average Pooling — mandatory 
    in_features = base.fc.in_features   # 512 for ResNet18
    base.fc = nn.Linear(in_features, n_classes)  # Custom classification head
    # fc is now trainable; rest frozen

    return base

tl_model = build_transfer_model(N_CLASSES, freeze_base=True).to(DEVICE)
